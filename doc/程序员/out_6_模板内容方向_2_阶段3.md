```python
# backend/app/api/timeline_api.py
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel

router = APIRouter()

class TimelineUpdateRequest(BaseModel):
    emphasis_words: list[str]
    timeline_data: dict

@router.put("/scenes/{scene_id}/timeline")
async def update_timeline(scene_id: str, request: TimelineUpdateRequest):
    """更新场景的时间轴数据"""
    scene = await Scene.get(scene_id)
    if not scene:
        raise HTTPException(status_code=404, detail="Scene not found")

    scene.emphasis_words = request.emphasis_words
    scene.timeline_data = request.timeline_data
    await scene.save()

    return {
        "success": True,
        "scene_id": scene_id,
        "updated_at": scene.updated_at
    }

@router.post("/scenes/{scene_id}/preview")
async def preview_scene(scene_id: str, start_time: float = 0, end_time: float = None):
    """生成场景预览视频"""
    scene = await Scene.get(scene_id)
    if not scene:
        raise HTTPException(status_code=404, detail="Scene not found")

    # 生成预览视频（快速渲染，低分辨率）
    preview_url = await preview_service.generate_preview(
        scene, start_time, end_time
    )

    return {
        "preview_url": preview_url,
        "duration": end_time - start_time if end_time else scene.duration_sec
    }
```

**做什么**：

**产出**：
```typescript
// frontend/src/stores/timelineStore.ts
import { create } from 'zustand';

interface Keyframe {
  id: string;
  time: number;
  element: string;
  action: 'pop' | 'fade_in' | 'slide_in';
  duration: number;
}

interface TimelineStore {
  keyframes: Keyframe[];
  selectedKeyframeId: string | null;
  isPlaying: boolean;
  currentTime: number;

  // Actions
  setKeyframes: (keyframes: Keyframe[]) => void;
  updateKeyframe: (id: string, updates: Partial<Keyframe>) => void;
  selectKeyframe: (id: string | null) => void;
  setPlaying: (playing: boolean) => void;
  setCurrentTime: (time: number) => void;
}

export const useTimelineStore = create<TimelineStore>((set) => ({
  keyframes: [],
  selectedKeyframeId: null,
  isPlaying: false,
  currentTime: 0,

  setKeyframes: (keyframes) => set({ keyframes }),
  updateKeyframe: (id, updates) =>
    set((state) => ({
      keyframes: state.keyframes.map((kf) =>
        kf.id === id ? { ...kf, ...updates } : kf
      ),
    })),
  selectKeyframe: (id) => set({ selectedKeyframeId: id }),
  setPlaying: (playing) => set({ isPlaying: playing }),
  setCurrentTime: (time) => set({ currentTime: time }),
}));
```

```typescript
// frontend/src/components/TimelineEditor/TimelineEditor.tsx
import { useRef, useEffect } from 'react';
import WaveSurfer from 'wavesurfer.js';
import { useTimelineStore } from '@/stores/timelineStore';
import { KeyframeTrack } from './KeyframeTrack';
import { TimelineRuler } from './TimelineRuler';

interface Props {
  sceneId: string;
  duration: number;
  audioUrl: string;
}

export function TimelineEditor({ sceneId, duration, audioUrl }: Props) {
  const waveformRef = useRef<HTMLDivElement>(null);
  const wavesurfer = useRef<WaveSurfer | null>(null);
  const { keyframes, setCurrentTime, isPlaying } = useTimelineStore();

  // 初始化音频波形
  useEffect(() => {
    if (waveformRef.current) {
      wavesurfer.current = WaveSurfer.create({
        container: waveformRef.current,
        waveColor: '#ddd',
        progressColor: '#3b82f6',
        height: 80,
        cursorWidth: 2,
      });
      wavesurfer.current.load(audioUrl);

      // 监听播放进度
      wavesurfer.current.on('audioprocess', (time) => {
        setCurrentTime(time);
      });
    }

    return () => wavesurfer.current?.destroy();
  }, [audioUrl]);

  return (
    <div className="timeline-editor">
      <div className="waveform-container" ref={waveformRef} />
      <TimelineRuler duration={duration} />
      <KeyframeTrack duration={duration} keyframes={keyframes} />
    </div>
  );
}
```

### Step 4：关键帧拖拽功能
**做什么**：
- 创建 KeyframeMarker 组件
- 实现拖拽调整时间点
- 防抖保存到后端

**产出**：
```typescript
// frontend/src/components/TimelineEditor/KeyframeMarker.tsx
import { useDrag } from 'react-dnd';
import { useTimelineStore } from '@/stores/timelineStore';
import { debounce } from 'lodash';

interface Props {
  keyframe: Keyframe;
  duration: number;
  onDrag: (newTime: number) => void;
}

export function KeyframeMarker({ keyframe, duration, onDrag }: Props) {
  const { selectKeyframe, selectedKeyframeId } = useTimelineStore();
  const isSelected = selectedKeyframeId === keyframe.id;

  // 拖拽逻辑
  const handleDrag = (e: React.DragEvent) => {
    const rect = e.currentTarget.parentElement!.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const newTime = (x / rect.width) * duration;
    onDrag(Math.max(0, Math.min(newTime, duration)));
  };

  // 防抖保存
  const debouncedSave = debounce((time: number) => {
    // 调用 API 保存
    fetch(`/api/scenes/${sceneId}/timeline`, {
      method: 'PUT',
      body: JSON.stringify({ timeline_data: {...} })
    });
  }, 300);

  return (
    <div
      className={`keyframe-marker ${isSelected ? 'selected' : ''}`}
      style={{ left: `${(keyframe.time / duration) * 100}%` }}
      draggable
      onDrag={handleDrag}
      onClick={() => selectKeyframe(keyframe.id)}
    >
      <div className="marker-label">{keyframe.element}</div>
    </div>
  );
}
```

**验证方式**：
- 拖拽关键帧，检查是否能调整时间
- 检查是否有防抖（300ms 后才保存）

### Step 5：重点词选择器
**做什么**：
- 创建 EmphasisWordSelector 组件
- 点击词语标记为重点词
- 最多选择 3 个

**产出**：
```typescript
// frontend/src/components/TimelineEditor/EmphasisWordSelector.tsx
import { useState } from 'react';

interface Props {
  voiceover: string;
  emphasisWords: string[];
  onEmphasisWordsChange: (words: string[]) => void;
}

export function EmphasisWordSelector({ voiceover, emphasisWords, onEmphasisWordsChange }: Props) {
  // 简单分词（按空格和标点）
  const words = voiceover.split(/[\s,，。.!！?？]+/).filter(w => w.length > 0);

  const toggleWord = (word: string) => {
    if (emphasisWords.includes(word)) {
      // 取消选中
      onEmphasisWordsChange(emphasisWords.filter(w => w !== word));
    } else if (emphasisWords.length < 3) {
      // 选中（最多 3 个）
      onEmphasisWordsChange([...emphasisWords, word]);
    }
  };

  return (
    <div className="emphasis-word-selector">
      <div className="label">点击选择重点词（最多 3 个）：</div>
      <div className="words">
        {words.map((word, i) => (
          <span
            key={i}
            className={`word ${emphasisWords.includes(word) ? 'selected' : ''}`}
            onClick={() => toggleWord(word)}
          >
            {word}
          </span>
        ))}
      </div>
    </div>
  );
}
```

**验证方式**：
- 点击词语，检查是否高亮
- 尝试选择超过 3 个，检查是否限制
- 再次点击，检查是否取消选中

### Step 6：动效选择器
**做什么**：
- 创建 EffectSelector 组件
- 显示 3 种动效选项
- 点击应用到选中的关键帧

**产出**：
```typescript
// frontend/src/components/TimelineEditor/EffectSelector.tsx
import { useTimelineStore } from '@/stores/timelineStore';

const EFFECTS = [
  { value: 'pop', label: 'Pop', description: '弹出效果' },
  { value: 'fade_in', label: 'Fade In', description: '淡入效果' },
  { value: 'slide_in', label: 'Slide In', description: '滑入效果' },
];

export function EffectSelector() {
  const { selectedKeyframeId, keyframes, updateKeyframe } = useTimelineStore();
  const selectedKeyframe = keyframes.find(kf => kf.id === selectedKeyframeId);

  if (!selectedKeyframe) {
    return <div className="effect-selector">请选择一个关键帧</div>;
  }

  const handleEffectChange = (action: string) => {
    updateKeyframe(selectedKeyframeId!, { action: action as any });
  };

  return (
    <div className="effect-selector">
      <div className="label">动效类型：</div>
      <div className="effects">
        {EFFECTS.map(effect => (
          <button
            key={effect.value}
            className={`effect-btn ${selectedKeyframe.action === effect.value ? 'active' : ''}`}
            onClick={() => handleEffectChange(effect.value)}
          >
            <div className="effect-label">{effect.label}</div>
            <div className="effect-desc">{effect.description}</div>
          </button>
        ))}
      </div>
    </div>
  );
}
```

**验证方式**：
- 选中一个关键帧
- 点击不同的动效选项
- 检查是否应用到关键帧

```python
# backend/app/services/preview_service.py
class PreviewService:
    async def generate_preview(self, scene: Scene, start_time: float = 0, end_time: float = None):
        """生成预览视频（快速渲染）"""
        # 使用低分辨率、低质量快速渲染
        preview_path = await render_service.render(
            scene,
            resolution="480p",
            quality="low",
            start_time=start_time,
            end_time=end_time
        )

        # 上传到 CDN
        preview_url = await cdn_service.upload(preview_path)

        # 缓存 10 分钟
        await cache.set(f"preview:{scene.id}", preview_url, ttl=600)

        return preview_url
```

**产出（前端）**：
```typescript
// frontend/src/components/TimelineEditor/PreviewPanel.tsx
import { useState } from 'react';

interface Props {
  sceneId: string;
}

export function PreviewPanel({ sceneId }: Props) {
  const [previewUrl, setPreviewUrl] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);

  const handlePreview = async () => {
    setLoading(true);
    try {
      const response = await fetch(`/api/scenes/${sceneId}/preview`, {
        method: 'POST'
      });
      const data = await response.json();
      setPreviewUrl(data.preview_url);
    } catch (error) {
      console.error('预览生成失败', error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="preview-panel">
      <button onClick={handlePreview} disabled={loading}>
        {loading ? '生成中...' : '预览'}
      </button>
      {previewUrl && (
        <video src={previewUrl} controls autoPlay />
      )}
    </div>
  );
}
```

## 5. 关键风险与应对

### 风险 1：前端编辑器性能差
**影响**：拖拽卡顿，用户体验差
**应对**：
- 使用虚拟滚动（关键帧 > 50 个）
- 防抖处理（300ms）
- 限制关键帧数量 < 20 个

### 风险 2：预览生成太慢
**影响**：用户等待时间长
**应对**：
- 降低预览分辨率（480p）
- 只渲染关键帧附近 ±1 秒
- 缓存预览结果（10 分钟）

### 风险 3：用户不使用编辑器
**影响**：开发浪费
**应对**：
- 如果使用率 < 20%，停止开发
- 分析原因：是否太复杂？是否不需要？
